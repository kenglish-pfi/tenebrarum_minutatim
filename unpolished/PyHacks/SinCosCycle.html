def sin4(X):
    list0 = []
    list1 = []
    list2 = []
    list3 = []
    for x in X:
        list0.append( math.sin(x) )
        list1.append( math.sin(math.pi/2.0 - x) )      # note: negative, sin(-x) = -sin(x)
        list2.append( math.sin(math.pi + x) )
        list3.append( math.sin(3.0*math.pi/2.0 - x) )
    return list0, list1, list2, list3

def cos4(X):
    list0 = []
    list1 = []
    list2 = []
    list3 = []
    for x in X:
        list0.append( math.cos(x) )
        list1.append( math.cos(math.pi/2.0 + x) )     # note: all positive ... cos(-x) = cos(x)
        list2.append( math.cos(math.pi + x) )
        list3.append( math.cos(3.0*math.pi/2.0 + x) )
    return list0, list1, list2, list3
